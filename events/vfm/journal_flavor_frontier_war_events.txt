journal_flavor_frontier_war_events.1 = { 
	type = country_event
	placement = ROOT

	title = journal_flavor_frontier_war_events.1.t
	desc = {
		triggered_desc = {
			desc = journal_flavor_frontier_war_events.1.d_WAS
			trigger = {
				ROOT = c:WAS
			}
		}
		triggered_desc = {
			desc = journal_flavor_frontier_war_events.1.d_NSW
			trigger = {
				ROOT = c:NSW
			}
		}
	}
	flavor = journal_flavor_frontier_war_events.1.f

	event_image = {
		video = "gfx/event_pictures/southamerica_war_civilians.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	dlc = dlc_vfm

	duration = 1

	immediate = {
		ROOT = {
			save_scope_as = frontier_war_country_scope
			set_variable = {
				name = frontier_var
				value = 0
			}
		}
		c:NNG = {
			save_scope_as = frontier_war_noongar_scope
		}
	}

	option = {
		name = journal_flavor_frontier_war_events.1.a
		default_option = yes
		add_journal_entry = { type = je_frontier_wars }
		
		custom_tooltip = {
			text = ast_frontier_1_tt
			set_variable = {
				name = frontier_var
				value = 1
			}
		}
    }
	option = {
		name = journal_flavor_frontier_war_events.1.b
		add_journal_entry = { type = je_frontier_wars }
		
		custom_tooltip = {
			text = ast_frontier_1_tt

}

#South Australia
journal_flavor_frontier_war_events.3 = { 
	type = country_event
	placement = ROOT

	title = journal_flavor_frontier_war_events.3.t
	desc = journal_flavor_frontier_war_events.3.d
	flavor = journal_flavor_frontier_war_events.3.f

	event_image = {
		video = "gfx/event_pictures/africa_construction_colony.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	dlc = dlc_vfm

	duration = 1

	trigger = {
		THIS = c:NSW
		NOT = {
			exists = c:SAS
		}
	}

	immediate = {
        s:STATE_SOUTH_AUSTRALIA = {
			set_owner_of_provinces = {
				country = c:NSW
				provinces = { xC58F87 xD0A0D0 xE5B909 x4064E2 xD53092 x90DF50 xC7963D x239C93 }
			}
        }
	}

	after = {
		create_country = {
            tag = SAS
            origin = c:WAS
            state = s:STATE_SOUTH_AUSTRALIA.region_state:NSW
		}
		s:STATE_SOUTH_AUSTRALIA = {
			remove_claim = c:NSW
			add_claim = c:SAS
	    }
	    if = {
	    	limit = { has_variable = wanna_south_ast }
	    	remove_variable = wanna_south_ast
			play_as = c:SAS
	    }
	    c:GBR = {
	        create_diplomatic_pact = {
	            country = c:SAS
	            type = chartered_company
	        }	
	    }	
	    s:STATE_SOUTH_AUSTRALIA.region_state:SAS = {
			create_pop = {
				pop_type = peasants
				culture = british
				size = 21000
			}
			create_pop = {
				pop_type = peasants
				culture = aborigine
				size = 9000
			}
	    }
	    c:SAS = {
			add_journal_entry = { type = je_frontier_wars }
	    	add_modifier = {
	    		name = colonial_boom
	    		years = 10
	    	}
	    	add_modifier = {
	    		name = terra_nullius
	    	}
	    	add_modifier = {
	    		name = settler_colony
	    	}
			create_character = {
				template = AST_john_hindmarsh
			}	
	
			create_character = {
				template = AST_robert_gouger
			}
			
			create_character = {
				template = AST_george_strickland_kensington
			}
			
			create_character = {
				template = AST_george_fife_angas
			}
			
			create_character = {
				template = AST_charles_howard
			}
	
			create_character = {
				template = AST_william_light
			}
	    }
	}

	option = {
		name = journal_flavor_frontier_war_events.3.a
		default_option = yes
		custom_tooltip = ast_frontier_2_tt
		hidden_effect = {
			s:STATE_SOUTH_AUSTRALIA.region_state:NSW = {
				create_building = {
					building = building_port
					level = 1
				}
			}
		}
		ai_chance = {
			base = 100
		}
	}
	option = {
		name = journal_flavor_frontier_war_events.3.b
		custom_tooltip = ast_frontier_2_tt
		hidden_effect = {
			s:STATE_SOUTH_AUSTRALIA.region_state:NSW = {
				create_building = {
					building = building_port
					level = 1
				}
			}
        	set_variable = {
        		name = wanna_south_ast
        		value = 0
        	}
		}
    }
}

#Aboriginal Smallpox Epidemic
journal_flavor_frontier_war_events.5 = { 
	type = country_event
	placement = ROOT

	title = journal_flavor_frontier_war_events.5.t
	desc = journal_flavor_frontier_war_events.5.d
	flavor = journal_flavor_frontier_war_events.5.f

	event_image = {
		video = "gfx/event_pictures/unspecific_sick_in_hospital.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	dlc = dlc_vfm

	duration = 1

	trigger = {
		any_neighbouring_state = {
			owner = {
				country_has_primary_culture = cu:aborigine
				NOR = {
					has_variable = aboriginal_smallpox_var
					has_variable = aboriginal_tb_var
					has_variable = aboriginal_flu_var
					has_variable = aboriginal_measles_var
				}
			}
		}
	}

	cooldown = {
		years = 2
	}

	immediate = {
		random_neighbouring_state = {
			limit = {
				owner = {
					country_has_primary_culture = cu:aborigine
					NOT = { has_variable = aboriginal_sickness_var }
				}
			}
			save_scope_as = aboriginal_sick_state_scope
			owner = {
				save_scope_as = aboriginal_sick_country_scope
				random_list = {
					1 = {
						set_variable = {
							name = aboriginal_smallpox_var
							value = yes
							years = 2
						}
					}
					1 = {
						set_variable = {
							name = aboriginal_tb_var
							value = yes
							years = 2
						}
					}
					1 = {
						set_variable = {
							name = aboriginal_flu_var
							value = yes
							years = 2
						}
					}
					2 = {
						set_variable = {
							name = aboriginal_measles_var
							value = yes
							years = 2
						}
					}
				}
			}
		}
	}

	option = {
		name = journal_flavor_frontier_war_events.5.a
		default_option = yes
		scope:aboriginal_sick_state_scope = {
			add_modifier = {
				name = aboriginal_foreign_epidemic
				years = 1
			}
		}
	}
	option = {
		name = journal_flavor_frontier_war_events.5.b

		scope:aboriginal_sick_state_scope = {
			owner = {
				change_tension = {
					country = ROOT
					value = -10
				}
			}
			add_modifier = {
				name = aboriginal_foreign_epidemic
				months = 8
				divide = 2
			}
		}
	}
}

journal_flavor_frontier_war_events.6 = { 
	type = country_event
	placement = ROOT

	title = journal_flavor_frontier_war_events.6.t
	desc = journal_flavor_frontier_war_events.6.d
	flavor = journal_flavor_frontier_war_events.6.f

	event_image = {
		video = "gfx/event_pictures/southamerica_war_civilians.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	dlc = dlc_vfm

	duration = 1

	trigger = {
		any_scope_state = {
			is_split_state = yes
			any_neighbouring_state = {
				owner = {
					country_has_primary_culture = cu:aborigine
				}
			}
		}
	}

	cooldown = {
		years = 1
	}

	immediate = {
		random_scope_state = {
			limit = {
				is_split_state = yes
				any_neighbouring_state = {
					owner = {
						country_has_primary_culture = cu:aborigine
					}
				}
			}
			save_scope_as = massacre_location_state_scope
			random_neighbouring_state = {
				limit = {
					owner = {
						country_has_primary_culture = cu:aborigine
					}
				}
				owner = {
					save_scope_as = massacre_location_country_scope
				}
			}
		}
	}

	option = {
		name = journal_flavor_frontier_war_events.6.a
		default_option = yes
		scope:massacre_location_state_scope = {
			kill_population_percent_in_state = {
				percent = 0.05
				culture = cu:aborigine
			}
		}
		scope:massacre_location_country_scope = {
			change_tension = {
				country = ROOT
				value = 10
			}
		}
	}
}
