namespace = frontier_war_events

#Frontier Wars Intro
frontier_war_events.1 = { 
	type = country_event
	placement = ROOT

	title = frontier_war_events.1.t
	desc = {
		triggered_desc = {
			desc = frontier_war_events.1.d_WAS
			trigger = {
				ROOT = c:WAS
			}
		}
		triggered_desc = {
			desc = frontier_war_events.1.d_NSW
			trigger = {
				ROOT = c:NSW
			}
		}
	}
	flavor = frontier_war_events.1.f

	event_image = {
		video = "gfx/event_pictures/southamerica_war_civilians.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	dlc = dlc_vfm

	duration = 1

	immediate = {
		ROOT = {
			save_scope_as = frontier_war_country_scope
			set_variable = {
				name = frontier_var
				value = 0
			}
		}
		c:NNG = {
			save_scope_as = frontier_war_noongar_scope
		}
	}

	option = {
		name = frontier_war_events.1.a
		default_option = yes
		add_journal_entry = { type = je_frontier_wars }
		
		custom_tooltip = {
			text = ast_frontier_1_tt
			set_variable = {
				name = frontier_var
				value = 1
			}
		}
    }
	option = {
		name = frontier_war_events.1.b
		add_journal_entry = { type = je_frontier_wars }
		
		custom_tooltip = {
			text = ast_frontier_1_tt
			set_variable = {
				name = frontier_var
				value = 2
			}
		}
    }
}
