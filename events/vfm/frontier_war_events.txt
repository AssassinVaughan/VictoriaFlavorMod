#Frontier Wars Intro
frontier_war_events.1 = { 
	type = country_event
	placement = ROOT

	title = frontier_war_events.1.t
	desc = {
		triggered_desc = {
			desc = frontier_war_events.1.d_WAS
			trigger = {
				ROOT = c:WAS
			}
		}
		triggered_desc = {
			desc = frontier_war_events.1.d_NSW
			trigger = {
				ROOT = c:NSW
			}
		}
	}
	flavor = frontier_war_events.1.f

	event_image = {
		video = "gfx/event_pictures/southamerica_war_civilians.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	dlc = dlc_vfm

	duration = 1

	immediate = {
		ROOT = {
			save_scope_as = frontier_war_country_scope
			set_variable = {
				name = frontier_var
				value = 0
			}
		}
		c:NNG = {
			save_scope_as = frontier_war_noongar_scope
		}
	}

	option = {
		name = frontier_war_events.1.a
		default_option = yes
		add_journal_entry = { type = je_frontier_wars }
		
		custom_tooltip = {
			text = ast_frontier_1_tt
			set_variable = {
				name = frontier_var
				value = 1
			}
		}
    }
	option = {
		name = frontier_war_events.1.b
		add_journal_entry = { type = je_frontier_wars }
		
		custom_tooltip = {
			text = ast_frontier_1_tt
			set_variable = {
				name = frontier_var
				value = 2
			}
		}
    }
}

frontier_war_events.3 = { 
	type = country_event
	placement = ROOT

	title = frontier_war_events.3.t
	desc = frontier_war_events.3.d
	flavor = frontier_war_events.3.f

	event_image = {
		video = "gfx/event_pictures/southamerica_aristocrats.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	dlc = dlc_vfm

	duration = 1

	trigger = {
		THIS = c:NSW
		NOT = {
			exists = c:SAS
		}
	}

	after = {
		create_country = {
            tag = SAS
            origin = c:WAS
            state = s:STATE_SOUTH_AUSTRALIA.region_state:NSW
		}
		s:STATE_SOUTH_AUSTRALIA = {
			remove_claim = c:NSW
			add_claim = c:SAS
	    }
	    if = {
	    	limit = { has_variable = wanna_south_ast }
	    	remove_variable = wanna_south_ast
			play_as = c:SAS
	    }
	    c:GBR = {
	        create_diplomatic_pact = {
	            country = c:SAS
	            type = chartered_company
	        }	
	    }	
	    s:STATE_SOUTH_AUSTRALIA.region_state:SAS = {
			create_building = {
				building = building_port
				level = 1
			}
			create_building = {
				building = building_livestock_ranch
				level = 1
			}
			create_pop = {
				pop_type = peasants
				culture = british
				size = 21000
			}
			create_pop = {
				pop_type = peasants
				culture = aborigine
				size = 9000
			}
	    }
	    c:SAS = {
			add_journal_entry = { type = je_frontier_wars }
	    	add_modifier = {
	    		name = colonial_boom
	    		years = 10
	    	}
	    	add_modifier = {
	    		name = terra_nullius
	    	}
	    	add_modifier = {
	    		name = settler_colony
	    	}
			create_character = {
				template = AST_john_hindmarsh
			}	
	
			create_character = {
				template = AST_robert_gouger
			}
			
			create_character = {
				template = AST_george_strickland_kensington
			}
			
			create_character = {
				template = AST_george_fife_angas
			}
			
			create_character = {
				template = AST_charles_howard
			}
	
			create_character = {
				template = AST_william_light
			}
	    }
	}

	option = {
		name = frontier_war_events.3.a
		default_option = yes
		custom_tooltip = ast_frontier_2_tt
		hidden_effect = {
        	s:STATE_SOUTH_AUSTRALIA = {
				set_owner_of_provinces = {
					country = c:NSW
					provinces = { xC58F87 xD0A0D0 xE5B909 x4064E2 xD53092 x90DF50 xC7963D x239C93 }
				}
        	}
		}
		ai_chance = {
			base = 100
		}
	}
	option = {
		name = frontier_war_events.3.b
		custom_tooltip = ast_frontier_2_tt
		hidden_effect = {
        	s:STATE_SOUTH_AUSTRALIA = {
				set_owner_of_provinces = {
					country = c:NSW
					provinces = { xC58F87 xD0A0D0 xE5B909 x4064E2 xD53092 x90DF50 xC7963D x239C93 }
				}
        	}
        	set_variable = {
        		name = wanna_south_ast
        		value = 0
        	}
		}
    }
}
