journal_flavor_germany.1 = {
	type = country_event
	event_image = {
		video = "unspecific_trains"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	duration = 3
	placement = scope:german_railroad_end
	title = journal_flavor_germany.1.t
	desc = journal_flavor_germany.1.d
	flavor = journal_flavor_germany.1.f

	trigger = {
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_BRANDENBURG
			}
			save_scope_as = german_railroad_end
		}
		ig:ig_petty_bourgeoisie = {
			save_scope_as = bourgeoisie_ig
		}

		ig:ig_industrialists = {
			save_scope_as = industrialists_ig
		}
	}
	
	option = {
		name = journal_flavor_germany.1.a
		default_option = yes
		add_modifier = {
			name = generic_production_tech_research
			months = long_modifier_time
		}
		ig:ig_petty_bourgeoisie = {
			add_modifier = {
				name = symbolist_ig_good_modifier
				months = normal_modifier_time
			}
		}
	}
	option = {
		name = journal_flavor_germany.1.b
		add_modifier = {
			name = generic_production_tech_research
			months = long_modifier_time
		}
		ig:ig_industrialists = {
			add_modifier = {
				name = symbolist_ig_good_modifier
				months = normal_modifier_time
			}
		}
	}
}

journal_flavor_germany.3 = {
	type = country_event
	event_image = {
		video = "ep1_printing_press"
	}
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	duration = 3
	placement = ROOT
	title = journal_flavor_germany.3.t
	desc = journal_flavor_germany.3.d
	flavor = journal_flavor_germany.3.f

	trigger = {
	}

	immediate = {
	    random_country = {
			limit = {
				OR = {
				    is_subject_of = ROOT
					is_in_customs_union_with = ROOT
				}
				NOR = {
					has_variable = var_member_two
					has_variable = var_member_three
				}
			}
			save_scope_as = scope_random_member_one
			set_variable = var_member_one
		}

		random_country = {
			limit = {
				OR = {
				    is_subject_of = ROOT
					is_in_customs_union_with = ROOT
				}
				NOR = {
					has_variable = var_member_one
					has_variable = var_member_three
				}
			}
			save_scope_as = scope_random_member_two
			set_variable = var_member_two
		}
		
		random_country = {
			limit = {
				OR = {
				    is_subject_of = ROOT
					is_in_customs_union_with = ROOT
				}
				NOR = {
					has_variable = var_member_one
					has_variable = var_member_two
				}
			}
			save_scope_as = scope_random_member_three
			set_variable = var_member_three
		}
	}
	
	option = {
		name = journal_flavor_germany.3.a
		default_option = yes
		scope:scope_random_member_one = {
		    set_owes_obligation_to = {
				country = ROOT
				setting = yes
			}
		}
		
		scope:scope_random_member_two = {
		    change_relations = {
				country = ROOT
				value = 20
			}
			add_modifier = {
				name = generic_production_tech_research
				months = long_modifier_time
			}
		}
		
		scope:scope_random_member_three = {
		    change_relations = {
				country = ROOT
				value = 20
			}
			add_modifier = {
				name = generic_production_tech_research
				months = long_modifier_time
			}
		}
	}
}
