je_puq_colonization = {
	icon = "gfx/interface/icons/event_icons/event_flags/event_chl_flag.dds"

	group = je_group_historical_content

	is_shown_when_inactive = {
		exists = c:CHL
		THIS = c:CHL
	}

	complete = {
		institution_investment_level = {
			institution = institution_colonial_affairs
			value >= 2
		}
	}

	on_complete = {
        trigger_event = { 
			id = journal_flavor_chile.3
		}
        add_modifier = {
            name = magellan_strait
        }				
	}

    timeout = 2555

    on_timeout = { s:STATE_ARAUCANIA = { remove_claim = ROOT } }

    should_be_pinned_by_default = yes
	weight = 100
}

je_ppp = {
	icon = "gfx/interface/icons/event_icons/event_flags/event_chl_flag.dds"

	group = je_group_historical_content

	is_shown_when_inactive = {
		exists = c:CHL
		THIS = c:CHL
	}

	immediate = {
		set_variable = {
			name = copy_eden_var
			value = 0
		}
	}

	complete = {
        country_or_subject_owns_entire_state_region = STATE_ARAUCANIA
		has_state_in_state_region = STATE_PATAGONIA
		exists = c:CHL
		this = c:CHL
		    OR = {
                any_neighbouring_state = {
			        owner = c:THL
		        }    
			    country_or_subject_owns_entire_state_region = STATE_PATAGONIA
			  }     			  
		NOT = { has_variable = patagonia_colonized_var }
		NOT = { has_journal_entry = je_patagonia }
	}

	on_complete = {
        trigger_event = { 
			id = journal_flavor_chile.3
		}
	}

    timeout = 1825

    on_timeout = { s:STATE_ARAUCANIA = { remove_claim = ROOT } }

	progressbar = yes	

    should_be_pinned_by_default = yes
	weight = 100
}
