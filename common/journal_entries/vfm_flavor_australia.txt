#The Frontier Wars
je_ast_frontier_wars = {
	icon = "gfx/interface/icons/event_icons/event_colonial.dds"
    group = je_group_historical_content

	scripted_button = je_send_settlers_button

	immediate = {
		ROOT = {
			# To use saved scopes in loc: JournalEntry.GetTopScope.sCountry('saved_scope_name') or SCOPE.sCountry('saved_scope_name')
			save_scope_as = frontier_war_country_je
		}
	}
	on_monthly_pulse = {
		random_events = {
			100 = 0
			#Historical
			15 = journal_flavor_frontier_war_events.3
			15 = journal_flavor_frontier_war_events.4
			#Generic
			2 = journal_flavor_frontier_war_events.5
			3 = journal_flavor_frontier_war_events.6
			3 = journal_flavor_frontier_war_events.7
			#Confederations
			3 = journal_flavor_frontier_war_events.8
		}
	}
	complete = {
		custom_tooltip = {
			text = ast_frontier_1_tt
			sr:region_oceania = {
				any_scope_state = {
					count = all
					OR = {
						AND = {
							has_claim_by = ROOT
							owner = ROOT
							
						}
						NOT = {
							has_claim_by = ROOT
						}
					}
				}
			}
		}
		any_scope_state = {
			count = all
			is_under_colonization = no
		}
	}

	on_complete = {
		trigger_event = {
			id = journal_flavor_frontier_war_events.2
		}
	}

    should_be_pinned_by_default = yes
	weight = 20
}

#The Southern Prison
je_convict_system = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"
	group = je_group_internal_affairs

	#timeout = 7300
	scripted_progress_bar = convict_system_progress_bar

	#scripted_button = je_import_convicts_button
	scripted_button = je_sway_settlers_button
	scripted_button = je_reform_freedom_button
	scripted_button = je_end_assignment_button

	immediate = {
		ROOT = {
			save_scope_as = je_convict_country_scope
			set_variable = {
				name = convict_progress_var
				value = 0
			}
			set_variable = {
				name = swaying_settlers_var
				value = 1
			}
		}
	}

	on_weekly_pulse = {
		events = {
			hidden_ast_events.1
		}
		effect = {
			if = {
				limit = {
					has_variable = importing_convicts_var
				}
				set_variable = {
					name = convict_progress_var
					value = -0.5
				}
			}
			if = {
				limit = {
					has_variable = swaying_settlers_var
				}
				set_variable = {
					name = convict_progress_var
					value = 1
				}
			}
		}
	}
	on_monthly_pulse = {
		random_events = {
			100 = 0
			5 = flavor_australia_events.3 #City protest
			5 = flavor_australia_events.6 #Government Lobby 1
			5 = flavor_australia_events.7 #Government Lobby 2
			5 = flavor_australia_events.8 #Squatters Sabotage Reform
		}
	}
	on_yearly_pulse = {
		events = {
			#Key Events
			flavor_australia_events.4
		}
	}

	#on_timeout = {
	#	trigger_event = { id = flavor_australia_events.3 }
	#}

	complete = {
		NOT = {
			has_law = law_type:law_debt_slavery 
		}
	}


	on_complete = {
		trigger_event = {
			id = flavor_australia_events.2
			days = 0
		}
	}

    should_be_pinned_by_default = yes
	weight = 20
}
