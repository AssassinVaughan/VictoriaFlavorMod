# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law

law_decentralized = {
	group = lawgroup_centralization
	
	icon = "gfx/interface/icons/law_icons/devolved.dds"
	
	progressiveness = -100

	on_activate = {
	}
	
	modifier = {
		country_bureaucracy_mult = -0.2
		country_bureaucracy_investment_cost_factor_mult = -0.1
		state_radicals_from_sol_change_mult = -0.2
		country_farmers_pol_str_mult = 0.25
		country_aristocrats_pol_str_mult = 0.5
		country_authority_add = -200
	}
	
	possible_political_movements = {
		law_confederated
		law_federal
	}

	pop_support = {
		value = 0
		# Aristocrats and Rural Folk like regionalism
		add = {
			desc = "POP_ARISTOCRATS"			
			if = {
				limit = { 
					is_pop_type = aristocrats
				}
				value = 0.75
			}
		}
		add = {
			desc = "POP_PEASANTS"
			if = {
				limit = {
					is_pop_type = peasants
				}
				value = 0.5
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = -25
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}
