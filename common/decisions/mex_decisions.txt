mex_purchase_ross = {
	is_shown = {
		exists = c:MEX
		exists = c:RUS
		c:MEX = ROOT
		is_subject = no
		NOT = {
			c:RUS = { has_state_in_state_region = STATE_CALIFORNIA }
		}
	}

	possible = {
		in_default = no
		taking_loans = no
		is_at_war = no
		c:RAC = { has_state_in_state_region = STATE_CALIFORNIA }
		c:RUS = {
			relations:root >= 0
		}		
	}

	when_taken = {
		custom_tooltip = {
			text = mex_purchase_ross_tt
			}
		hidden_effect = {
		    s:STATE_CALIFORNIA = {
			    set_owner_of_provinces = {
				    country = c:MEX
				    provinces = { x108000 }
			        }
		        }		
		    }
		add_modifier = {
			name = ross_mex_big_money
			months = 7
			is_decaying = no
		}
		c:RAC = {
		add_modifier = {
			name = ross_rus_big_money
			months = 7
			is_decaying = no
		}		    		    
	}
}	

	ai_chance = {
		base = 50	
		modifier = {
			trigger = { gold_reserves > 10000 }
			add = 10
		}		
		modifier = {
			trigger = { 
				gold_reserves > 0
				weekly_net_fixed_income > 6000
			}
			add = 10
		}
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}
	}
}
